
# 1. Retrieve all orders with order_status "shipped"
db.orders.find({ order_status: "shipped" }).pretty()

# 2. Update the total_amount of the order with order_id: 1 to 70000
db.orders.updateOne(
  { order_id: 1 },
  { $set: { total_amount: 70000 } }
)

# Verify update
db.orders.find({ order_id: 1 }).pretty()

# 3. Delete the order with order_id: 4
db.orders.deleteOne({ order_id: 4 })

# Verify deletion
db.orders.find().pretty()

# 4. Retrieve the order with customer_name "Alice Johnson"
db.orders.find({ customer_name: "Alice Johnson" }).pretty()

# 5. Update the order_status of the order with order_id: 2 to "delivered"
db.orders.updateOne(
  { order_id: 2 },
  { $set: { order_status: "delivered" } }
)

# Verify update
db.orders.find({ order_id: 2 }).pretty()

# 6. Retrieve all orders with total_amount >= 15000
db.orders.find({ total_amount: { $gte: 15000 } }).pretty()
